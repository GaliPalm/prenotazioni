    const google = {

      script: {

        run: {

          withSuccessHandler: function(successCallback) {

            return {

              withFailureHandler: function(failureCallback) {

                return {

                  // Mappatura delle funzioni del tuo backend

                  getAppData: function() {

                    eseguiChiamataAPI("getAppData", {}, successCallback, failureCallback);

                  },

                  bookEvent: function(sheetName, parentName, studentName, school) {

                    eseguiChiamataAPI("bookEvent", {sheetName, parentName, studentName, school}, successCallback, failureCallback);

                  },

                  cancelBooking: function(sheetName, parentName, studentName) {

                    eseguiChiamataAPI("cancelBooking", {sheetName, parentName, studentName}, successCallback, failureCallback);

                  }

                };

              }

            };

          }

        }

      }

    };


    // Funzione helper che invia i dati a Google Apps Script in modo sicuro

    function eseguiChiamataAPI(azione, dati, onSuccess, onFailure) {

      const payload = { action: azione, ...dati };

      

      fetch(URL_WEB_APP, {

        method: "POST",

        mode: "cors",

        headers: { "Content-Type": "text/plain;charset=utf-8" },

        body: JSON.stringify(payload)

      })

      .then(response => response.json())

      .then(risultato => {

        if (onSuccess) onSuccess(risultato);

      })

      .catch(errore => {

        console.error("Errore API:", errore);

        if (onFailure) onFailure(errore);

      });

    }
